git branch			show the branch you're on
git status			show the branch you're on
git branch --all		all the branches (including remote ones)

git branch dev			create new branch
git switch -c dev		... v2.23+
git checkout dev		switch to another branch
git switch dev			... v2.23+
git checkout -b dev		create & siwtch. same as "git branch dev && git checkout dev"
git checkout --orphan tmp	create new branch in a `git init`-like state on a non-new repository
git branch -d dev		delete branch
git branch -D dev		force delete branch. shortcut for: --delete --force
git pull origin dev		pull from branch
git push -u origin dev		push new branch to remote

git push -d origin <branch_name>			delete remote branch
git branch -d <branch_name>				delete local branch

git branch --set-upstream-to=origin/master master	set upstream
