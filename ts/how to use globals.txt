Two ways:

-------------------------------------------------------------------------------
declare them at the begining of each file:

declare var u: any, newPubSub: any;

-------------------------------------------------------------------------------
declare them in an app-level .d.ts file

global.d.ts:

declare var newPubSub: any;
declare var u: any;
declare var UIkit: any;
declare var noUiSlider: any;
declare var vis: any;
