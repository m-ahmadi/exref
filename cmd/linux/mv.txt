mv file.txt other/		move file
mv file.txt file2.txt		rename file

mv folder another/folder	move folder
mv folder folder2		rename folder

mv * other/			move all files & folders
mv *.csv other/			... csv files only

--------------------------------------------------
mv --help

-i	interactive, ask before overwrite
-f	force, do not ask before overwriting
-v	verbose
