dir			current dir
dir node_modules	specified dir
dir *.js		all js files
dir /b			bare format (no heading/information/summary)
dir /s			all subdirs (recursive)
dir /a:d		only files with specified attributes. d: dir, h: hidden, r: readonly, s: system
dir /a:-d 		only files
dir /a:d /b		only dirs
dir ..\ /p		pause on screen end
dir /a /s		everything. any sys,hidden,... file in all dirs
dir file.txt /s		search for file everywhere under current dir
